#include OjAdapter

def id_path(id, *parts)
  # Using 2/2/2 split.
  # See https://github.com/cyber-dojo/id-split-timer
  args = ['katas', id[0..1], id[2..3], id[4..5]]
  args += parts.map(&:to_s)
  File.join(*args)
end
